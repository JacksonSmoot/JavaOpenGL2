
    private static void catchErr(String ifFailed, int code){
        if(code < 0){
            System.err.println(ifFailed+" | \n\t"+ffErr(code));
            System.exit(code);
        }
    }

    private static String ffErr(int code) {
        BytePointer buffer = new BytePointer(1024);
        avutil.av_strerror(code, buffer, 1024);
        return buffer.getString();
    }

    public static void saveFrame(AVFrame pFrame, int width, int height, int iFrame) {
        String filename = "frame" + iFrame + ".ppm";

        // PPM header (P6 = binary RGB)
        String header = "P6\n" + width + " " + height + "\n255\n";

        // pFrame must be RGB24 packed in data[0]
        BytePointer data0 = pFrame.data(0);
        int linesize0 = pFrame.linesize(0);

        if (data0 == null || data0.isNull()) {
            System.err.println("Frame has no data[0]");
            return;
        }

        try (FileOutputStream out = new FileOutputStream(filename)) {
            out.write(header.getBytes(StandardCharsets.US_ASCII));

            // Write each row: width * 3 bytes
            for (int y = 0; y < height; y++) {
                long rowOffset = (long) y * (long) linesize0;
                // Copy exactly width*3 bytes from this row
                byte[] row = new byte[width * 3];
                data0.position(rowOffset).get(row, 0, row.length);
                out.write(row);
            }
        } catch (IOException e) {
            System.err.println("Cannot open/write file: " + filename);
            e.printStackTrace();
        } finally {
            // restore position so callers aren't surprised
            data0.position(0);
        }
    }

    /**
     * Converts an AVFrame to a packed RGBA ByteBuffer.
     *
     * The returned buffer is native memory (memAlloc) and MUST be freed with memFree().
     *
     * @param frame input AVFrame (any pixel format)
     * @return RGBA ByteBuffer (size = width * height * 4)
     */
    public static ByteBuffer frameToRGBA(AVFrame frame) {
        int srcW = frame.width();
        int srcH = frame.height();
        int srcFormat = frame.format();

        if (srcW <= 0 || srcH <= 0) {
            throw new IllegalArgumentException("Invalid frame size");
        }

        // Allocate output buffer (native memory)
        int bytes = srcW * srcH * 4;
        ByteBuffer rgbaBuffer = memAlloc(bytes);

        // Create sws context (convert to RGBA)
        SwsContext sws = swscale.sws_getContext(
                srcW, srcH, srcFormat,
                srcW, srcH, avutil.AV_PIX_FMT_RGBA,
                swscale.SWS_BILINEAR,
                null, null, (double[]) null
        );

        if (sws == null) {
            memFree(rgbaBuffer);
            throw new RuntimeException("sws_getContext failed");
        }

        // Prepare destination pointers
        BytePointer dstPtr = new BytePointer(rgbaBuffer);

        PointerPointer<BytePointer> dstData = new PointerPointer<>(4);
        dstData.put(0, dstPtr);
        dstData.put(1, null);
        dstData.put(2, null);
        dstData.put(3, null);

        IntPointer dstLinesize = new IntPointer(4);
        dstLinesize.put(0, srcW * 4);
        dstLinesize.put(1, 0);
        dstLinesize.put(2, 0);
        dstLinesize.put(3, 0);

        // Convert
        swscale.sws_scale(
                sws,
                frame.data(),
                frame.linesize(),
                0,
                srcH,
                dstData,
                dstLinesize
        );

        // Clean up sws
        swscale.sws_freeContext(sws);
        dstLinesize.close();
        dstData.close();

        rgbaBuffer.position(0);
        return rgbaBuffer;
    }

    private static int ffmpeg_main()
    {
        int outputFormat = AV_PIX_FMT_RGBA; // AV_PIX_FMT_RGB24

        // Register all formats and codecs
        // Now not useful anymore since version 4.0
        //av_register_all();
        Path sample_60fps = Path.of("/Users/jacksonsmoot/com/jrs/dev/JavaOpenGL2/src/main/resources/sample_60fps_720p.mp4");
        // Open video file
        AVFormatContext pFormatCtx = avformat.avformat_alloc_context();
        if (pFormatCtx == null) {
            System.err.println("Could not allocate memory for format context");
            return -1;
        }

        if (avformat.avformat_open_input(pFormatCtx, sample_60fps.toString(), null, null) != 0)
        {
            System.err.println("Cannot open file");
            return -1; // Couldn't open file
        }

        // Retrieve stream information
        if (avformat.avformat_find_stream_info(pFormatCtx, (PointerPointer)null) < 0)
        {
            System.err.println("Could not find stream information");
            return -1; // Couldn't find stream information
        }

        AVCodec pCodec = null;
        AVCodecParameters pCodecParams = null;

        // Find the first video stream
        int videoStream = -1;
        int i;

        for (i = 0; i < pFormatCtx.nb_streams(); i++)
        {
            if (pFormatCtx.streams(i).codecpar().codec_type() == AVMEDIA_TYPE_VIDEO)
            {
                videoStream = i;
                pCodecParams = pFormatCtx.streams(i).codecpar();
                pCodec = avcodec_find_decoder(pCodecParams.codec_id());
                break; // We want only the first video stream, the leave the other stream that might be present in the file
            }
        }

        if (videoStream == -1)
        {
            System.err.println("Did not find a video stream");
            return -1;
        }

        // Get a pointer to the codec context for the video stream
        AVCodecContext pCodecCtx = avcodec_alloc_context3(pCodec);
        if (pCodecCtx == null)
        {
            System.err.println("Could not allocate video codec context");
            return -1;
        }

        // Fill the codec context from the codec parameters values
        if (avcodec_parameters_to_context(pCodecCtx, pCodecParams) < 0)
        {
            System.err.println("Failed to copy codec params to codec context");
            return -1;
        }

        if (avcodec_open2(pCodecCtx, pCodec, (PointerPointer)null) < 0)
        {
            System.err.println("failed to open codec");
            return -1;
        }

        // Allocate video frame
        AVFrame pFrame = av_frame_alloc();
        if (pFrame == null)
        {
            System.err.println("Could not allocate video frame");
            return -1;
        }

        AVPacket pPacket = av_packet_alloc();
        if (pPacket == null)
        {
            System.err.println("Failed to allocate memory for AVPacket");
            return -1;
        }

        // Allocate an AVFrame structure
        AVFrame pFrameRGB = av_frame_alloc();
        if(pFrameRGB == null)
        {
            System.err.println("Could not allocate output video frame");
            return -1;
        }

        SwsContext swsCtx =
            swscale.sws_getContext
                    (
                            pCodecCtx.width(),
                            pCodecCtx.height(),
                            pCodecCtx.pix_fmt(),
                            pCodecCtx.width(),
                            pCodecCtx.height(),
                            outputFormat,
                            SWS_BILINEAR,
                            (SwsFilter) null,
                            (SwsFilter) null,
                            (DoublePointer) null
                    );

        int frames_to_process = 5;
        i = 0;

        // Read frames and save first five frames to disk
        while (avformat.av_read_frame(pFormatCtx, pPacket) >= 0 && i < frames_to_process)
        {
            // Is this a packet from the video stream?
            if (pPacket.stream_index() == videoStream)
            {
                // Decode video frame
                if (avcodec_send_packet(pCodecCtx, pPacket) < 0)
                {
                    System.err.println("Error while sending packet to decoder.");
                    return -1;
                }

                int frameFinished = 1;
                // Did we get a video frame?
                while (frameFinished >= 0 && i < frames_to_process)
                {
                    frameFinished = avcodec_receive_frame(pCodecCtx, pFrame);
                    // These two return values are special and mean there is no output
                    // frame available, but there were no errors during decoding
                    if (frameFinished == AVERROR_EAGAIN() || frameFinished == AVERROR_EOF)
                    {
                        break;
                    }
                    else if (frameFinished < 0)
                    {
                        System.err.println("Error during decoding");
                        return -1;
                    }

                    if (frameFinished >= 0)
                    {
                        if (avutil.av_image_alloc(pFrameRGB.data(), pFrameRGB.linesize(), pCodecCtx.width(), pCodecCtx.height(), outputFormat, 1) < 0)
                        {
                            System.err.println("Could not allocate output frame");
                            return -1;
                        }

                        // Convert the image from its native format to RGB
                        swscale.sws_scale
                                (
                                        swsCtx,
                                        pFrame.data(),
                                        pFrame.linesize(),
                                        0,
                                        pCodecCtx.height(),
                                        pFrameRGB.data(),
                                        pFrameRGB.linesize()
                                );

                        // Save the frame to disk
                        saveFrame(pFrameRGB, pCodecCtx.width(), pCodecCtx.height(), ++i);
                    }
                }
            }

            // Free the packet that was allocated by av_read_frame
            av_packet_unref(pPacket);
        }

        sws_freeContext(swsCtx);

        // Free the YUV frame
        av_frame_free(pFrame);

        // Free the RGB image
        av_frame_free(pFrameRGB);

        // Close the codec
        avcodec_free_context(pCodecCtx);

        // Close the video file
        avformat_close_input(pFormatCtx);

        return 0;
    }
